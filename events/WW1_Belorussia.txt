# GERMAN ANSWER
######################
country_event = {
	title = ww1_belorussia.4.t
	desc = ww1_belorussia.4.d
	id = ww1_belorussia.4
	picture = GFX_report_event_spr_civil_war_faction_defeated

	is_triggered_only = yes
	
	option = {
		name = ww1_belorussia.4.a
		puppet = BLR
		hidden_effect = {
			BLR = {
				set_politics = {
					ruling_party = neutrality
					last_election = "1908.11.1"
					election_frequency = 48
					elections_allowed = no
				}
				set_popularities = {
					democratic = 23
					fascism = 10
					communism = 5
					neutrality = 62
				}
				set_cosmetic_tag = BLR_WHR
				if = {
					limit = { has_government = neutrality }
					if = {
						limit = { date < 1929.4.20 }
						BLR_genrikh = {
							add_country_leader_role = {
								promote_leader = yes
								country_leader = {
									ideology = semi_constitutional_monarchy
									traits = { german_agent foreign_leader }							
								}
							}
						}
						GER = {
							GER_albert_wilhelm_heinrich = {
								set_character_name = BLR_genrikh
								set_nationality = BLR
							}
						}
					}	
					else = {
						BLR_uladzimir = {
							add_country_leader_role = {
								promote_leader = yes
								country_leader = {
									ideology = semi_constitutional_monarchy							
								}
							}
						}
						GER = {
							retire_character = GER_albert_wilhelm_heinrich
						}
					}
					set_party_name = {
						ideology = neutrality 
						long_name = GER_neutrality_party 
						name = GER_neutrality_party
					}
				}
			}
		}
		ai_chance = { factor = 25 }
	}
	option = {
		name = ww1_belorussia.4.b
		give_guarantee = BLR
		BLR = { add_stability = 0.1 }
		ai_chance = { factor = 5 }
	}
}

# DEATH OF HEINRICH

country_event = {
	id = ww1_belorussia.6
	title = ww1_belorussia.6.t
	desc = ww1_belorussia.6.d
	fire_only_once = yes
	trigger = {
		original_tag = BLR
		has_country_leader = {
			character = BLR_genrikh
			ruling_only = yes
		}
		date > 1929.4.1
	}
	mean_time_to_happen = { days = 20 }
	immediate = {
		hidden_effect = {
			every_possible_country = {
				limit = {
					has_character = GER_albert_wilhelm_heinrich
				}
				retire_character = GER_albert_wilhelm_heinrich
			}
			retire_character = BLR_genrikh
			BLR_uladzimir = {
				add_country_leader_role = {
					promote_leader = yes
					country_leader = {
						ideology = semi_constitutional_monarchy
					}
				}
			}
		}
	}
	option = {
		name = ww1_belorussia.6.a
		ai_chance = { factor = 100 }
		effect_tooltip = {
			retire_character = BLR_genrikh
		}
	}
}

# German Kingdom collapses
country_event = {
	id = ww1_belorussia.8
	title = ww1_belorussia.8.t
	desc = ww1_belorussia.8.d
	fire_only_once = yes
	trigger = {
		original_tag = BLR
		has_cosmetic_tag = BLR_WHR
		has_global_flag = GER_capitulation
	}
	mean_time_to_happen = { days = 10 }
	immediate = {
		hidden_effect = {
			every_possible_country = {
				limit = {
					has_character = GER_albert_wilhelm_heinrich
				}
				retire_character = GER_albert_wilhelm_heinrich
			}
			retire_character = BLR_genrikh
			BLR_uladzimir = {
				add_country_leader_role = {
					promote_leader = yes
					country_leader = {
						ideology = semi_constitutional_monarchy
					}
				}
			}
		}
	}
	option = {
		name = ww1_belorussia.8.a
		if = {
			limit = {
				is_subject = yes
				OVERLORD = { has_country_flag = great_war_loser }
			}
			OVERLORD = { end_puppet = ROOT }
		}
		drop_cosmetic_tag = yes
		meta_effect = {
			text = {
				set_politics = {
					ruling_party = neutrality
					long_name = BLR_neutrality_party
					name = BLR_neutrality_party
					last_election = "[CURRENT_DATE]"
					elections_allowed = no
				}
			}
			CURRENT_DATE = "[?global.date.GetDateStringNoHour]"
		}
		set_popularities = {
			fascism = 2
			communism = 41
			democratic = 4
			neutrality = 53
		}
		hidden_effect = {
			GER_albert_wilhelm_heinrich = {
				set_nationality = GER
			}
			BLR_genrikh = {
				remove_country_leader_role = {
					ideology = semi_constitutional_monarchy
				}
			}
			BLR_uladzimir = {
				remove_country_leader_role = {
					ideology = semi_constitutional_monarchy
				}
			}
		}
	}
	option = {
		name = ww1_belorussia.8.b
		trigger = { is_ai = no }
		add_stability = -0.10
		add_political_power = -170
		add_popularity = {
			ideology = communism
			popularity = 0.15
		}
	}
}
