country_event = {
	id = ww1_ukraine.2
	title = ww1_ukraine.2.t
	desc = ww1_ukraine.2.d
	picture = GFX_report_event_yugoslavian_cavalry
	fire_only_once = yes
	is_triggered_only = yes
	option = {
		name = ww1_ukraine.2.a
		every_state = {
			limit = {
				OR = {
					is_claimed_by = UKR
					is_core_of = UKR
				}
				OR = {
					is_owned_by = RUS
					is_owned_by = SOV
					is_owned_by = BLR
					is_owned_by = POL
				}	
			}
			transfer_state_to = UKR
			remove_core_of = RUS
			remove_core_of = SOV
		}
		hidden_effect = {
			if = {
				limit = {
					has_civil_war = yes
				}
				set_country_flag = ukrainian_civil_war_again
				random_other_country = {
					limit = {
						original_tag = UKR
						has_war_with = PREV
					}
					every_owned_state = {
						limit = {
							is_controlled_by = PREV.PREV
						}
						set_state_owner_to = PREV.PREV
					}
					every_controlled_state = {
						limit = {
							is_owned_by = PREV.PREV
						}
						set_state_owner_to = PREV
					}
					every_state = {
						limit = {
							is_owned_by = PREV.PREV
							region = 40
						}
						set_state_owner_to = PREV
					}
				}
			}
		}
		GER = {
			if = {
				limit = {
					has_subject_autonomy_feature = yes
				}
				set_autonomy = {
					target = UKR
					autonomous_state = autonomy_tgwr_informal_protectorate
				}
				UKR = { set_country_flag = protectorate } 
				else = {
					puppet = UKR	
					UKR = { add_ideas = protectorate } 
				}
			}
		}
		UKR = {
			set_popularities = {
				democratic = 35
				fascism = 10
				communism = 24
				neutrality = 31
			}
			set_politics = {
				ruling_party = neutrality
				elections_allowed = no
			}
			promote_character = UKR_pavlo_skoropadskyi
			set_country_flag = ukrainian_hetmanate
			hidden_effect = {
				set_cosmetic_tag = UKR_GETMAN
				if = {
					limit = {
						has_country_flag = ukrainian_civil_war_again
					}
					random_other_country = {
						limit = {
							original_tag = UKR
							is_in_faction = no
						}
						declare_war_on = {
							target = ROOT
							type = civil_war
						}
						add_civil_war_target = ROOT
					}
					clr_country_flag = ukrainian_civil_war_again
				}
			}
		}
		RUS = {
			remove_unit_leader = 76234
			RUS_aleksandr_ragoza = {
				promote_leader = yes
				set_nationality = UKR				
			}
			RUS_fyodor_keller = {
				promote_leader = yes
				set_nationality = UKR				
			}
		}	
		ai_chance = { factor = 90 }
	}
	option = {
		name = ww1_ukraine.2.b
		GER = { country_event = { id = ww1_ukraine.3 days = 1 } }
		ai_chance = { 
			factor = 10
			modifier = {
				factor = 0
				is_historical_focus_on = yes
			} 
		}
	}
}

		tag = UKR
		has_government = neutrality
		has_global_flag = great_war_ends

			OR = {
				has_country_flag = great_war_loser

			}

			limit = {
				is_in_faction_with = GER

			}

		}
		set_politics = {
			ruling_party = democratic
			elections_allowed = yes
		}
		set_popularities = {
			democratic = 50

			communism = 40

		}

    	clr_country_flag = ukrainian_hetmanate

    		ai_chance = { 
    			factor = 100
		}

			retire_character = RUS_fyodor_keller
		}
	}

					target = ROOT
