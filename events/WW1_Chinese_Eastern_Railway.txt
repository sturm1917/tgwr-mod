add_namespace = ww1_cer_rcw

#Harbin Soviet Uprising
country_event = {
	id = ww1_cer_rcw.1
	title = ww1_cer_rcw.1.t
	desc = ww1_cer_rcw.1.d
	picture = rusrevolt_6
	fire_only_once = yes
	is_triggered_only = yes
	trigger = {
		has_idea = CER_harbin_soviet
		is_subject = no
		NOT = {
			has_cosmetic_tag = CER_provisional_russia
			has_cosmetic_tag = CER_high_commission
		}
	}
	immediate = {
		hidden_effect = {
			remove_ideas = CER_harbin_soviet
			SOV = {
				diplomatic_relation = {
					country = ROOT
					relation = military_access
					active = no
				}
				diplomatic_relation = {
					country = ROOT
					relation = guarantee
					active = no
				}
			}
			start_civil_war = {
				ideology = communism
				size = 0.1
				states = { 910 } 
				keep_unit_leaders_trigger = {
					always = yes
				}
				give_military_access = SOV
				set_cosmetic_tag = CER_communism
				SOV = {
					give_guarantee = PREV
				}
				every_other_country = {
					limit = {
						has_military_access_to = CER
					}
					diplomatic_relation = {
						country = PREV
						relation = military_access
						active = yes
					}
				}
			}	
		}
	}
	option = {
		name = ww1_cer_rcw.1.a
		country_event = {
			id = ww1_cer_rcw.2
			days = 14
		}
	}
	option = {
		name = ww1_cer_rcw.1.b
		random_other_country = {
			limit = {
				original_tag = CER
				has_government = communism
			}
			SOV = {
				if = {
					limit = {
						has_subject_autonomy_feature = yes
					}
					set_autonomy = {
						target = PREV
						autonomous_state = autonomy_tgwr_concession
						freedom_level = 0
						end_civil_wars = no
						end_wars = no
					}
					else = {
						puppet = {
							target = PREV
							end_civil_wars = no
							end_wars = no
						}	
						PREV = { add_ideas = concession_state }
					}
				}
			}
			change_tag_from = ROOT
		}
		trigger = {
			is_ai = no
		}
	}
}
