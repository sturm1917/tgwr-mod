	id = ww1_vietnam.5
	title = ww1_vietnam.5.t
	desc = ww1_vietnam.5.d
	fire_only_once = yes
	trigger = {
		tag = GER
		has_global_flag = frankfurt_treaty
		OR = {
			FRA = {
				has_civil_war = yes
			}
			any_country = {
				has_country_flag = FRA_nationalist_france_flag
				has_war_with = FRA
			}
		}
		any_state = {
			OR = {
				is_owned_by = FRA
				if = {
					limit = {
						VIN = { has_country_flag = great_war_loser }
					}
					is_owned_by = VIN
				}
				if = {
					limit = {
						LAO = { has_country_flag = great_war_loser }
					}
					is_owned_by = LAO
				}
				if = {
					limit = {
						CAM = { has_country_flag = great_war_loser }
					}
					is_owned_by = CAM
				}
			}
			OR = {
				is_core_of = VIN
				is_core_of = LAO
				is_core_of = CAM
			}
		}
	}
	mean_time_to_happen = {
		days = 3
	}
	option = {
		name = ww1_vietnam.5.a
		ai_chance = {
			factor = 100
		}
		every_state = {
			limit = {
				OR = {
					is_owned_by = FRA
					if = {
						limit = {
							VIN = { has_country_flag = great_war_loser }
						}
						is_owned_by = VIN
					}
					if = {
						limit = {
							LAO = { has_country_flag = great_war_loser }
						}
						is_owned_by = LAO
					}
					if = {
						limit = {
							CAM = { has_country_flag = great_war_loser }
						}
						is_owned_by = CAM
					}
				}
				OR = {
					is_core_of = VIN
					is_core_of = LAO
					is_core_of = CAM
				}
			}
			OST = {
				transfer_state = PREV
				add_state_core = PREV
			}
		}
		if = {
			limit = {
				MAL = {
					exists = yes
					is_subject = no
				}
			}
			OST = {
				annex_country = {
					target = MAL
					transfer_troops = no
				}
				add_state_core = 333
				add_state_core = 336
				set_capital = { state = 336 }
			}
		}
		every_state = {
			limit = {
				is_owned_by = GER
				OR = {
					state = 336
					state = 333
					state = 334
				}
			}
			OST= {
				transfer_state = PREV
				if = {
					limit = {
						owns_state = 333
					}
					add_state_core = 333
				}
				if = {
					limit = {
						owns_state = 336
					}
					add_state_core = 336
					set_capital = { state = 336 }
				}
			}
		}
		GER = {
			if = {
				limit = {
					has_dlc = "Together for Victory"
				}
				set_autonomy = {
					target = OST
					autonomy_state = autonomy_colony
				}
			}
			else = {
				puppet = OST
			}
			every_unit_leader = {
				limit = {
					has_id = 19141
				}
				set_nationality = OST
			}
		}
		if = {
			limit = {
				SIA = {
					exists = yes
					OR = {
						NOT = {
							has_country_flag = great_war_loser
							GER = { has_completed_focus = GER_ostasien }
						}
						has_country_flag = great_war_winner
					}
				}
			}
			every_state = {
				limit = {
					is_owned_by = OST
					OR = {
						state = 819
						state = 822
					}
				}
				SIA = {
					transfer_state = PREV
				}
			}
		}
		OST = {
			add_ideas = GER_idea_colonial_exploitation_interwar
			add_ideas = GER_dysfunctional_government
			add_ideas = OST_colonial_question
			add_ideas = OST_Protecting_GER_interest
		}
	}

			factor = 0
		}
