					OR = {
						is_core_of = VIN

						is_core_of = LAO
					}
				}
				remove_claim_by = ROOT
				add_core_of = ROOT
			}
		}
		remove_ideas = standing_alone
		random_other_country = {
			limit = {
				original_tag = FRA

					has_government = communism
					has_country_flag = FRA_socialist_france_flag
				}
				has_war = no
				is_subject = no
			}
			if = {
				limit = {
					has_dlc = "Together for Victory"

				}

					puppet = ROOT

			}

					name = FRA_albert_pierre_sarraut
