	fire_only_once = yes
	trigger = {
		tag = BAK
		has_civil_war = yes
		is_subject = no
		has_government = communism
	}
	mean_time_to_happen = { days = 30 }
	option = {
		name = ww1_azerbaijan.2.a
		every_other_country = {
			limit = {
				has_russian_civil_war = yes
				has_war_with = ROOT
			}
			white_peace = ROOT
		}
		set_politics = {
			ruling_party = neutrality
			elections_allowed = no
		}	
		add_popularity = {
			ideology = neutrality
			popularity = 0.2
		}
		division_template = {
			name = "Bakı Ordusu"
			# is_locked = yes
			division_names_group = AZR_INF_03
			template_counter = 69
			regiments = {
				infantry = { x = 0 y = 0 }
				infantry = { x = 0 y = 1 }
				infantry = { x = 1 y = 0 }
			}
			support = {
				artillery = { x = 0 y = 0 }
			}
		}
		
		every_country_division = {
			change_division_template = {
				division_template = "Bakı Ordusu"
			}
		}
		delete_unit_template_and_units = {
			division_template = "Bakının Qırmızı Ordusu"
		}
		hidden_effect = {
			if = {
				limit = {
					any_enemy_country = {
						original_tag = AZR
						is_subject = yes
						OVERLORD = {
							has_war_with = ROOT
							any_enemy_country = {
								original_tag = ENG
								is_subject = no
								is_in_faction = yes
								has_capitulated = no
								OR = {
									has_subject = PER
									any_owned_state = { is_core_of = PER }
								}
							}
						}
					}
				}
				country_event = {
					id = ww1_azerbaijan.4
					hours = 12
				}
			}
		}
	}
}
