# Remove Kingdom of Lithuania
####################
country_event = {
	id = ww1_lithuania.1
	title = ww1_lithuania.1.t
	desc = ww1_lithuania.1.d
	trigger = {
		tag = LIT
		has_cosmetic_tag = LIT_GER
		OR = {
			AND = {
				is_subject = yes
				OVERLORD = { has_country_flag = great_war_loser }
			}
			has_country_flag = great_war_loser
		}
	}
	mean_time_to_happen = {
		days = 5
	}
	fire_only_once = yes
	hidden = yes
	immediate = {
		hidden_effect = {
			if = {
				limit = {
					is_in_faction = yes
				}
				leave_faction = yes
			}
			every_other_country = {
				limit = {
					has_war_with = ROOT
				}
				white_peace = ROOT
			}
			if = {
				limit = {
					is_subject_of = GER
				}
				GER = {
					end_puppet = LIT
				}
			}
		}
	}
	option = {
		name = ww1_lithuania.1.a
		drop_cosmetic_tag = yes
		LIT_antanas_smetona = {
			promote_character = authoritarian_democracy
		}
		hidden_effect = {
			meta_effect = {
				text = {
					set_politics = {
						ruling_party = neutrality
						long_name = LIT_neutrality_party_long
						name = LIT_neutrality_party
						last_election = "[CURRENT_DATE]"
						elections_allowed = yes
					}
				}
				CURRENT_DATE = "[?global.date.GetDateStringNoHour]"
			}
			set_popularities = {
				democratic = 30
				communism = 20
				fascism = 0
				neutrality = 50
			}
			add_ideas = limited_conscription
			add_ideas = export_focus
			add_ideas = civilian_economy
		}
	}
}

# Death of Mindaugas II
####################
country_event = {
	id = ww1_lithuania.2
	title = ww1_lithuania.2.t
	desc = ww1_lithuania.2.d
	picture = GFX_report_event_europe_funeral
	trigger = {
		original_tag = LIT
		has_country_leader = {
			character = LIT_mindaugas_ii
			ruling_only = no
		}
		date > 1928.3.0
	}
	mean_time_to_happen = {
		days = 24
	}
	fire_only_once = yes
	option = {
		name = ww1_lithuania.2.a
		hidden_effect = { retire_character = GER_wilhelm_karl_von_urach }
		retire_character = LIT_mindaugas_ii
		LIT_vytautas_ii = {
			set_character_name = LIT_vytautas_ii
			promote_leader = yes
			add_country_leader_role = {
				country_leader = {
					ideology = semi_constitutional_monarchy
					traits = { LIT_king_of_lithuania POL_soldier_king german_agent }
				}
			}
			promote_character = semi_constitutional_monarchy
		}
	}
}

# Monarch arrived - they are told to speak Lithunanians, staying in Lithuania only
country_event = {
	id = ww1_lithuania.6
	hidden = yes
	trigger = {
		tag = LIT
		has_country_leader_ideology = semi_constitutional_monarchy
		has_cosmetic_tag = LIT_GER
		OR = {
			has_country_leader = {
				character = LIT_mindaugas_ii
			}
			has_country_leader = {
				character = LIT_vytautas_ii
			}
		}
		has_country_flag = great_war_winner
	}
	fire_only_once = yes
	immediate = {
		set_party_name = {
			ideology = neutrality
			long_name = LIT_house_of_urach
			name = LIT_house_of_urach
		}
		if = {
			limit = {
				has_character = LIT_mindaugas_ii
			}
			if = {
				limit = {
					has_character = GER_wilhelm_karl_von_urach
				}
				GER_wilhelm_karl_von_urach = {
					set_portraits = {
						civilian = {
							large = "gfx/leaders/LIT/LIT_mindaugas_ii_civilian.png"
						}
						army = {
							large = "gfx/leaders/LIT/LIT_mindaugas_ii.png"
						}
					}
					promote_leader = yes
				}
			}
			LIT_mindaugas_ii = {
				set_portraits = {
					civilian = {
						large = "gfx/leaders/LIT/LIT_mindaugas_ii_civilian.png"
					}
					army = {
						large = "gfx/leaders/LIT/LIT_mindaugas_ii.png"
					}
				}
				add_country_leader_trait = LIT_king_of_lithuania
				add_country_leader_trait = POL_soldier_king
			}
			LIT_vytautas_ii = {
				add_corps_commander_role = {
					skill = 1
					attack_skill = 1
					defense_skill = 1
					planning_skill = 1
					logistics_skill = 1
					traits={ politically_connected media_personality cavalry_officer trait_royal }
					visible = {
						NOT = { has_government = communism }
					}
				}
			}
		}
		else_if = {
			limit = {
				has_character = LIT_vytautas_ii
			}
			LIT_vytautas_ii = {
				set_character_name = LIT_vytautas_ii
				if = {
					limit = {
						is_unit_leader = yes
					}
					promote_leader = yes
					else = {
						add_field_marshal_role = {
							skill = 1
							attack_skill = 1
							defense_skill = 1
							planning_skill = 1
							logistics_skill = 1
							traits={ politically_connected media_personality cavalry_officer trait_royal }
							visible = {
								NOT = { has_government = communism }
							}
						}
					}
				}
				add_country_leader_trait = LIT_king_of_lithuania
				add_country_leader_trait = POL_soldier_king
			}
		}
	}
}

# Palanga
country_event = {
	id = ww1_lithuania.7
	title = ww1_lithuania.7.t
	desc = ww1_lithuania.7.d
	picture = GFX_report_event_generic_peaceful_annexation
	trigger = {
		owns_state = 946
		has_war = no
		LIT = {
			has_war = no
			is_neighbor_of = ROOT
			NOT = {
				has_war_with = ROOT
			}
			OR = {
				has_government = ROOT
				AND = {
					OR = {
						has_government = democratic
						has_government = neutrality
					}
					ROOT = {
						NOT = { has_government = communism }
					}
				}
			}
			OR = {
				is_subject = no
				is_subject_of = ROOT
				AND = {
					is_subject = yes
					OVERLORD = { has_subject = ROOT }
				}
			}
		}
	}
	fire_only_once = yes
	option = {
		name = ww1_lithuania.7.a
		effect_tooltip = {
			LIT = {
				transfer_state = 946
				add_opinion_modifier = {
					target = ROOT
					modifier = small_increase
				}
			}
		}
		hidden_effect = {
			LIT = {
				country_event = ww1_lithuania.8
			}
		}
	}
	option = {
		name = ww1_lithuania.7.b
		946 = {
			add_resistance = 10
		}
		ai_chance = { base = 0 }
	}
}
