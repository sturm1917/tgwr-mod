# TREATY OF TRIANON
#####################
country_event = {
	id = ww1_hungary.3
	title = ww1_hungary.3.t
	desc = ww1_hungary.3.d
	picture = GFX_report_event_hungary_southern_slovakia
	fire_only_once = yes
	is_triggered_only = yes
	option = {
		name = ww1_hungary.3.a
		add_ideas = HUN_treaty_of_triannon
		add_ideas = idea_army_restrictions
		# Reparations
		add_timed_idea = {
			idea = GER_idea_reparations
			days = 365
		}
		# Territories
		every_country = {
			limit = {
				has_country_flag = hungarian_romanian_war_winner
			}
			every_state = {
				limit = {
					is_core_of = PREV
					is_owned_by = HUN
				}
				PREV = {
					transfer_state = PREV
				}
			}
		}
		# Burgenland
		if = {
			limit = {
				owns_state = 147
			}
			random_other_country = {
				limit = {
					owns_state = 4
				}
				transfer_state = 147
			}
		}
		# Prekmurje
		if = {
			limit = {
				owns_state = 875
			}
			random_other_country = {
				limit = {
					owns_state = 102
				}
				transfer_state = 875
			}
		}
		# Remove claims
		every_state = {
			limit = {
				OR = {
					is_claimed_by = HUN
					is_core_of = HUN
				}
				NOT = {
					is_owned_by = HUN
				}
			}
			remove_claim_by = HUN
			remove_core_of = HUN
		}
		# Hungarian national cores
		every_state = {
			limit = {
				OR = {
					state = 45
					state = 83
					state = 147
					state = 664
					state = 778
					state = 875
				}
			}
			add_core_of = HUN
		}
	}
}

# Conditional surrender to Soviet Hungary
country_event = {
	id = ww1_hungary.8
	title = ww1_hungary.8.t
	desc = ww1_hungary.8.d
	picture = GFX_report_event_hungarian_soldiers
	is_triggered_only = yes
	option = {
		name = ww1_hungary.8.a
		FROM = {
			every_owned_state = {
				limit = {
					OR = {
						is_core_of = ROOT
						is_claimed_by = ROOT
					}
				}
				transfer_state_to = ROOT
			}
			every_other_country = {
				limit = {
					is_subject_of = ROOT
				}
				every_state = {
					limit = {
						is_owned_by = PREV.PREV
						OR = {
							is_core_of = PREV
							is_claimed_by = PREV
						}
					}
					transfer_state_to = PREV
				}
			}
			white_peace = ROOT
			hidden_effect = {
				if = {
					limit = {
						NOT = {
							has_government = communism
						}
					}
					add_popularity = {
						ideology = communism
						popularity = 0.10
					}
				}
			}
		}
		ai_chance = {
			base = 10
			modifier = {
				add = 40
				surrender_progress > 0.10
			}
			modifier = {
				add = 20
				any_other_country = {
					NOT = { original_tag = ROM }
					has_war_with = ROOT
				}
			}
			modifier = {
				add = 20
				has_war_support < 0.5
			}
		}
	}
	option = {
		name = ww1_hungary.8.b
		FROM = { country_event = ww1_hungary.9 }
		ai_chance = {
			base = 10
			modifier = {
				add = 20
				has_war_support > 0.5
			}
		}
	}
}
