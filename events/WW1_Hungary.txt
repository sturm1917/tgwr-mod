# TREATY OF TRIANON
#####################
country_event = {
	id = ww1_hungary.3
	title = ww1_hungary.3.t
	desc = ww1_hungary.3.d
	picture = GFX_report_event_generic_sign_treaty1
	fire_only_once = yes
	is_triggered_only = yes
	option = {
		name = ww1_hungary.3.a
		add_ideas = HUN_treaty_of_triannon
		add_ideas = idea_army_restrictions
		# Reparations
		add_timed_idea = {
			idea = GER_idea_reparations
			days = 365
		}
		# Territories
		every_country = {
			limit = {
				has_country_flag = hungarian_romanian_war_winner
			}
			every_state = {
				limit = {
					is_core_of = PREV
					is_owned_by = HUN
				}
				PREV = {
					transfer_state = PREV
				}
			}
		}
		# Burgenland
		if = {
			limit = {
				owns_state = 147
			}
			if = {
				limit = {
					country_exists = AUS
				}
				AUS = {
					add_state_core = 147
					transfer_state = 147
				}
				else_if = {
					limit = {
						country_exists = AUH
					}
					AUH = {
						add_state_core = 147
						transfer_state = 147
					}
				}
			}
		}
		# Remove claims
		every_state = {
			limit = {
				OR = {
					is_claimed_by = HUN
					is_core_of = HUN
				}
				NOT = {
					is_owned_by = HUN
				}
			}
			remove_claim_by = HUN
			remove_core_of = HUN
		}
		add_state_core = 45
		add_state_core = 83
		add_state_core = 147
		add_state_core = 664
		add_state_core = 778
	}
}
