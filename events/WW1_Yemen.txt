add_namespace = ww1_yemen

# End of Ottoman suzerainty over Najran
country_event = {
	id = ww1_yemen.1
	title = ww1_yemen.1.t
	desc = ww1_yemen.1.d
	picture = GFX_report_event_generic_peaceful_annexation
	fire_only_once = yes
	trigger = {
		tag = NAJ
		has_war = no
		OR = {
			AND = {
				is_subject = yes
				OVERLORD = {
					original_tag = TUR
					has_country_flag = great_war_loser
				}
			}
			is_subject = no
		}
	}
	mean_time_to_happen = { days = 0 }
	option = {
		name = ww1_yemen.1.a
		if = {
			limit = { is_subject = yes }
			OVERLORD = {
				end_puppet = ROOT
			}
		}
		if = {
			limit = {
				YEM = {
					exists = yes
					NOT = { has_country_flag = great_war_loser }
				}
			}
			# yemeni vassal
			YEM = {
				if = {
					limit = {
						ROOT = { owns_state = 930 }
					}
					transfer_state = 930
				}
				if = {
					limit = {
						has_subject_autonomy_feature = yes
					}
					set_autonomy = {
						target = ROOT
						autonomy_state = autonomy_dominion
					}
				}
				else = {
					puppet = ROOT
				}
			}
			add_ideas = arab_vassal_state
		}
	}
}
