#Sidney Street
news_event = {
	id = eng_news.21
	title = eng_news.21.t
	desc = eng_news.21.d
	picture = GFX_news_event_sidney_street
	fire_only_once = yes
	major = yes
	trigger = {
		date > 1911.01.03
		date < 1911.03.03
		ENG = {
			has_character = ENG_winston_churchill
			NOT = { has_government = communism }
		}
		NOT = {
			has_global_flag = sidney_street
		}
	}
	immediate = {
		hidden_effect = {
			set_global_flag = sidney_street
		}
	}
	option = {
		name = eng_news.21.a
		trigger = { 
			NOT = { 
				OR = { 
					original_tag = RUS
					original_tag = LAT
					is_subject_of = ENG
					AND = {
						is_major = yes
						NOT = {
							original_tag = ENG
						}
					}
				}
			} 
		}
		ai_chance = { factor = 100 }
	}
	option = {
		name = eng_news.21.b
		trigger = {
			OR = {
				is_subject_of = ENG 
				AND = { 
					is_major = yes 
					NOT = { 
						OR = { 
							original_tag = ENG 
							original_tag = RUS
							original_tag = LAT
						}
					} 
				} 
			} 
		}
		ai_chance = { factor = 100 }
	}
	option = {
		name = eng_news.21.c
		trigger = { OR = { original_tag = RUS original_tag = LAT } }
		ai_chance = { factor = 100 }
	}
}
