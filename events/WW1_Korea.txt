country_event = {
	title = ww1_korea.59.t
	desc = {
		text = ww1_korea.59.d
		trigger = {
			OR = {
				KOR = { is_subject_of = JAP }
				AND = {
					NOT = { country_exists = KOR }
					JAP = { owns_state = 800 }
				}
			}
		}
	}
	desc = {
		text = ww1_korea.59.d_independent
		trigger = {
			country_exists = KOR
			NOT = {
				KOR = { is_subject_of = JAP }
				JAP = { owns_state = 800 }
			}
		}
	}
	id = ww1_korea.59
	picture = GFX_report_event_generic_funeral
	fire_only_once = yes
	mean_time_to_happen = {
		days = 1
	}
	trigger = {
		if = {
			limit = {
				country_exists = KOR
			}
			tag = KOR
			else = {
				owns_state = 800
			}
		}
		KOR = {
			has_character = KOR_yi_hui
			NOT = { has_country_flag = emperor_gwangmu_is_dead }
		}
		date > 1919.1.21
	}
	immediate = {
		hidden_effect = {
			KOR = {
				if = {
					limit = { NOT = { has_country_flag = emperor_gwangmu_is_dead } }
					retire_character = KOR_yi_hui
					promote_character = KOR_yi_cheok
					if = {
						limit = {
							has_completed_focus = KOR_god_help_our_emperor
						}
						KOR_yi_cheok = {
							remove_country_leader_trait = inexperienced_monarch
						}
					}
					set_country_flag = emperor_gwangmu_is_dead
				}
			}
			if = {
				limit = {
					KOR = { is_subject_of = JAP }
				}
				JAP = { country_event = ww1_korea.59 }
			}
		}
	}
	option = {
		name = ww1_korea.59.a
		trigger = {
			tag = KOR
		}
		if = {
			limit = {
				has_idea = KOR_idea_gwangmu
			}
			swap_ideas = {
				remove_idea = KOR_idea_gwangmu
				add_idea = KOR_idea_yunghui
			}
		}
		effect_tooltip = {
			promote_character = KOR_yi_cheok
			if = {
				limit = {
					has_completed_focus = KOR_god_help_our_emperor
				}
				KOR_yi_cheok = {
					remove_country_leader_trait = inexperienced_monarch
				}
			}
		}
		hidden_effect = {
			if = {
				limit = {
					is_subject_of = JAP
				}
				country_event = ww1_japan.8
			}
		}
	}
	option = {
		name = ww1_korea.59.b
		trigger = {
			NOT = { tag = KOR }
		}
	}
}
