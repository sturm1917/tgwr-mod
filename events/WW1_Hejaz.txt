					target = PREV

						add_ideas = protectorate
					}
				}
			}

			limit = {
				is_neighbor_of = FROM

			}
			diplomatic_relation = {
				country = PREV
				relation = non_aggression_pact
				active = yes
			}
		}
		hidden_effect = {
			set_politics = {
				ruling_party = neutrality
				elections_allowed = no
			}
			set_popularities = {
				neutrality = 100
			}

		}
