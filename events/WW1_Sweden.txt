		}

				transfer_troops = yes
			}			
		}
		every_state = {
			limit = {
				is_core_of = ROOT
			}
			add_core_of = SWE
		}
		if = {
			limit = {
				is_ai = no
				SWE = {
					is_ai = yes
				}
			}
			SWE = { change_tag_from = ROOT }
		}
		news_event = {
			id = ww1_sweden.13
		}
	}
	#resist!
	option = {
		name = ww1_sweden.12.b

				factor = 0

				}

					has_war = no
				}
				is_in_faction = no
			}
		}
		SWE = {

				type = annex_everything
				target = ROOT
			}
		}
		every_state = {
