		}

	}
	mean_time_to_happen = {
		days = 365
	}
	immediate = {
		every_possible_country = {
			limit = {
				OR = {
					has_character = QIN_xuantong

			}

						}
					}
				}
			}
			if = {
				limit = {
					has_character = CHI_xuantong
				}

							small = "GFX_portrait_man_henry_pu_yi_emperor_small"
						}
					}
				}
			}
			if = {
				limit = {
					has_character = ZHC_xuantong
				}
				ZHC_xuantong = {
					set_portraits = {
