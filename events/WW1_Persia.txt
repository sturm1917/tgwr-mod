# 1921 Persian coup d'état
################################
country_event = {
	id = ww1_persia.6
	title = ww1_persia.6.t
	desc = ww1_persia.6.d
	picture = GFX_report_event_generic_defensive_preparations
	trigger = {
		tag = PER
		date > 1921.2.1
		has_war = no
		OR = {
			has_government = neutrality
			has_government = fascism
		}
		OR = {
			is_subject_of = GER
			is_subject_of = ENG
			AND = {
				is_subject = no
				any_state = {
					is_core_of = PER
					is_owned_by = ENG
				}
			}
		}
	}
	mean_time_to_happen = {
		days = 21
	}
	immediate = {
		hidden_effect = {
			if = {
				limit = {
					is_subject = yes
				}
				OVERLORD = {
					end_puppet = PREV
				}
			}
			every_state = {
				limit = {
					is_core_of = PER
					OR = {
						is_owned_by = ENG
						is_owned_by = RUS
					}
				}
				PER = {
					transfer_state = PREV
				}
			}
		}
	}
	option = {
		name = ww1_persia.6.a
		hidden_effect = {
			if = {
				limit = {
					has_idea = idea_victors_of_wwi
				}
				remove_ideas = idea_victors_of_wwi
			}
			if = {
				limit = {
					has_idea = idea_armistice
				}
				remove_ideas = idea_armistice
			}
			if = {
				limit = {
					has_idea = idea_army_restrictions
				}
				remove_ideas = idea_army_restrictions
			}
		}
		set_politics = {
			ruling_party = neutrality
			elections_allowed = no
		}
		PER_ahmad_shah_qajar = {
			remove_country_leader_role = {
				ideology = semi_constitutional_monarchy
			}
		}
		PER_reza_khan = {
			add_country_leader_role = {
				promote_leader = yes
				country_leader = {
					ideology = stratocracy
				}
			}
			add_country_leader_role = {
				promote_leader = yes
				country_leader = {
					ideology = despotism
				}
			}
		}
		add_political_power = -100
		add_stability = -0.1
		add_war_support = 0.1
		ai_chance = {
			factor = 100
		}
	}
	option = {
		name = ww1_persia.6.b
		start_civil_war = {
			ideology = fascism
			size = 0.5
			keep_political_leader = yes
			keep_political_party_members = yes
			set_country_flag = PER_pahlavi_flag
			set_cosmetic_tag = PER_gendarmerie
			PREV = {
				PER_reza_khan = {
					set_character_flag = PER_pahlavi_flag
					set_nationality = PREV.PREV
				}
				PER_abdollah_amir_tahmasebi = {
					set_character_flag = PER_pahlavi_flag
					set_nationality = PREV.PREV
				}
				PER_karim_buzarjomehri = {
					set_character_flag = PER_pahlavi_flag
					set_nationality = PREV.PREV
				}
			}
			keep_unit_leaders_trigger = {
				OR = {
					has_trait = trait_royal
					has_trait = trait_royalist
					has_trait = politically_connected
				}
			}
			PER_reza_khan = {
				add_country_leader_role = {
					promote_leader = yes
					country_leader = {
						ideology = stratocracy
					}
				}
				add_country_leader_role = {
					promote_leader = yes
					country_leader = {
						ideology = despotism
					}
				}
			}
		}
		add_political_power = -200
		add_stability = -0.2
		add_war_support = -0.1
		ai_chance = {
			factor = 0
		}
	}
}

	id = ww1_persia.7
	title = ww1_persia.7.t
	desc = ww1_persia.7.d
	picture = GFX_report_event_royal_parade
	trigger = {
		original_tag = PER
		is_subject = no
		has_war = no
		OR = {
			is_subject = no
			has_government = neutrality
		}
		date > 1925.12.1
		has_country_leader = {
			character = PER_reza_khan
			ruling_only = yes
		}
	}
	mean_time_to_happen = {
		days = 25
	}
	option = {
		name = ww1_persia.7.a
		hidden_effect = {
			set_popularities = {
				democratic = 25
				neutrality = 55
				fascism = 20
				communism = 0
			}
			set_politics = {
				ruling_party = neutrality
				elections_allowed = no
			}
			set_cosmetic_tag = PER_pahlavi
		}
		set_party_name = {
			ideology = neutrality
			long_name = PER_pahlavi_dynasty
			name = PER_pahlavi_dynasty
		}
		PER_reza_khan = {
			remove_country_leader_role = {
				ideology = despotism
			}
			set_character_name = PER_reza_shah_pahlavi
			add_country_leader_role = {
				promote_leader = yes
				country_leader = {
					ideology = autocracy
				}
			}
		}
	}
}
