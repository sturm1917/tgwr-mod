# Somalia
country_event = {
	id = ww1_somalia.2
	title = ww1_somalia.2.t
	desc = ww1_somalia.2.d
	picture = GFX_report_event_generic_conference
	fire_only_once = yes
	mean_time_to_happen = {
		days = 60
	}
	trigger = {
		tag = SOM
		NOT = {
			has_character = SOM_mohammed_abdullah_hassan
		}
		has_country_flag = great_war_winner
		is_subject = no
		has_war = no
	}
	option = {
		name = ww1_somalia.2.a
		add_stability = 0.10
		add_political_power = 150
		set_cosmetic_tag = SOM_somalia
		if = {
			limit = {
				has_country_leader = {
					character = SOM_maxamuud_cali_shiire
					ruling_only = yes
				}
			}
			SOM_maxamuud_cali_shiire = {
				add_country_leader_role = {
					promote_leader = yes
					country_leader = {
						ideology = autocracy
					}
				}
				add_unit_leader_trait = trait_royal
			}
		}
	}
}
