# Wallachia

country_event = {
	id = color.10
	
	trigger = {
		original_tag = ROM
		MOL = { exists = yes }
		NOT = {
			any_state = {
				is_core_of = MOL
				NOT = { 
					state = 80 
					state = 868
					state = 867
					state = 866
				}
				is_owned_by = ROM
			} 
		}
		NOT = {
			AND = {		
				TRA = { exists = no }
				any_state = {
					is_core_of = TRA
					NOT = { 
						state = 80 
						state = 868
						state = 867
						state = 866
					}
					is_owned_by = ROM
				}
			}
		}
		NOT = { has_cosmetic_tag = WLC }		
		has_civil_war = no
	}	
	mean_time_to_happen = {
		days = 0
	}

	hidden = yes

	immediate = {
		set_cosmetic_tag = WLC
	}
}
