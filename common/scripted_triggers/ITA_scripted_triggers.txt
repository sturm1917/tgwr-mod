		original_tag = EMI

		original_tag = SIC
		original_tag = TUS

	}
}

}
has_italian_civil_war = {
	custom_trigger_tooltip = {
		tooltip = has_italian_civil_war_tt
		OR = {
			ITA = {
				has_civil_war = yes
			}
			any_country = {
				is_italian_nation = yes
				any_enemy_country = {
					is_italian_nation = yes
				}
			}
		}
	}
}

			is_italian_nation = yes
		}
	}
}
