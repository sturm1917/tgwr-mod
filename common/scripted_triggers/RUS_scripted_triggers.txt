is_russian_tag = {
	OR = {
		original_tag = RUS
		original_tag = FER
		original_tag = CER
		is_cossack_tag = yes
	}
}
is_russian_core = {
	OR = {
		original_tag = RUS
		original_tag = FER
		original_tag = KUB
		original_tag = DON
		original_tag = ORE
		original_tag = UCH	
		original_tag = CER
	}
	any_other_country = {
		OR = {
			original_tag = RUS
			original_tag = FER
			original_tag = KUB
			original_tag = DON
			original_tag = ORE
			original_tag = UCH	
			original_tag = CER
		}
		PREV = { is_core_of = PREV }
	}
}
is_russia_a_monarchy = {
	custom_trigger_tooltip = {
		tooltip = is_russia_a_monarchy_tt
		OR = {
			has_country_leader  = {
				character = RUS_nikolai_ii
				ruling_only  = yes
			}
			has_country_leader  = {
				character = RUS_mikhail_ii
				ruling_only  = yes
			}
			has_idea = nicholas_ii
			has_idea = mikhail_ii
		}
		NOT = {
			has_cosmetic_tag = RUS_republic
			has_cosmetic_tag = RUS_prov
			has_cosmetic_tag = RUS_socialism
			has_cosmetic_tag = SOV
			has_cosmetic_tag = RUS_state
		}
	}
}
has_russian_civil_war = {
	custom_trigger_tooltip = {
		tooltip = has_russian_civil_war_tt
		OR = {
			RUS = {
				has_civil_war = yes
			}
			any_country = {
				is_russian_tag = yes
				has_war_with = SOV	
			}
		}	
	}
}
is_in_russian_civil_war = {
	custom_trigger_tooltip = {
		tooltip = has_russian_civil_war_tt
		OR = {
			AND = {
				is_russian_tag = yes
				has_civil_war = yes
			}
			AND = {
				is_russian_tag = yes
				any_enemy_country = {
					is_russian_tag = yes
				}
			}
		}
	}
}
is_cossack_tag = {
	OR = {
		original_tag = KUB
		original_tag = DON
		original_tag = ORE
		original_tag = UCH
		original_tag = TBK
		original_tag = AMU
		original_tag = TER
	}
}
