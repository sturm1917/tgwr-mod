ROM_prepare_for_great_war = {
	allowed = {
		tag = ROM
	}
	enable = {	
		NOT = {
			has_war_with = RUS
			has_war_with = BUL
			has_war_with = MOL
		}	
		OR = {
			date > 1914.1.1
			threat > 0.5
		}
		NOT = {
			has_country_leader = {
				character = ROM_carol_i
			}
			has_idea = ROM_carol_i
		}
	}
	abort = {
		date > 1918.1.1
	}
	ai_strategy = {
		type = force_build_armies
		value = 200
	}
	ai_strategy = {
		type = antagonize
		id = "AUH"
		value = 100
	}
}
