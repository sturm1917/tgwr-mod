BUL_defend_homeland = {
	allowed = {
		original_tag = BUL
	}
	enable = {
		has_global_flag = kis_war
		OR = {
			has_war_with = FRA
			has_war_with = GER
		}
		SER = {
			has_capitulated = yes
			has_war_with = ROOT
		}
		GRE = {
			is_neighbor_of = ROOT
			NOT = { has_war_with = ROOT }
		}
		NOT = {
			has_global_flag = great_war_ends
		}
	}
	abort_when_not_enabled = yes
	ai_strategy = {
		type = front_unit_request	
		strategic_region = 25
		value = 200
	}
	ai_strategy = {
		type = front_unit_request	
		tag = GRE
		value = 200
	}
	ai_strategy = {
		type = prepare_for_war
		id = GRE
		value = 200
	}
}
